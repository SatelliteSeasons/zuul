001package pkg_room;
002
003import java.util.HashMap;
004
005
006/**
007 * Classe ImageDirectionList - Une classe qui liste les 4 images de 4 directions différentes d'un lieu/room.
008 *
009 * @author Richard Ho
010 * @version 22/05/2023
011 */
012public class ImageDirectionList
013{
014    private HashMap<String,String> aImageDirectionList;
015    private String aImageName;
016
017    /**
018     * Constructeur d'objets de classe ImageDirectionList
019     * @param pImageName Le nom de l'image/fichier en String.
020     */
021    public ImageDirectionList( final String pImageName )
022    {
023        this.aImageDirectionList = new HashMap<String, String>();
024        this.aImageName = pImageName;
025        this.ajouterImage( "N", this.aImageName+"_N" );
026        this.ajouterImage( "E", this.aImageName+"_E" );
027        this.ajouterImage( "S", this.aImageName+"_S" );
028        this.ajouterImage( "W", this.aImageName+"_W" );
029    }//ImageDirectionList()
030    
031    /**
032     * Méthode qui permet d'ajouter une image à la liste.
033     * @param pDirection La direction que vise l'image avec un seul caractère en String.
034     * @param ImageName Le nom de l'image en String.
035     */
036    public void ajouterImage( final String pDirection, final String ImageName )
037    {
038        this.aImageDirectionList.put(pDirection, ImageName);
039    }//ajouterImage()
040    
041    /**
042     * Méthode qui permet d'obtenir le nom de l'image.
043     * @param pDirection La direction en String et un seul caractère.
044     * @return Le nom de l'image en String.
045     */
046    public String getImage( final String pDirection )
047    {
048        return this.aImageDirectionList.get( pDirection );
049    }//getImage()
050    
051}//ImageDirectionList
