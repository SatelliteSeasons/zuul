329    /**
330     * Méthode qui affiche la description du lieu courant et les différents direction possible des
331     * prochains lieux.
332     */
333    public void printLocationInfo()
334    {
335        if( this.aPlayer.getCurrentRoom().getImageName()!=null){
336                this.aGui.showImage( this.aPlayer.getCurrentRoom().getImageHashMap().getImage( this.aPlayer.getDirection() ) );
337        }
338        this.aGui.println( this.aPlayer.getCurrentRoom().getLongDescription() );
339        this.printFirstTalk();
340        if(aPersonnage.get(0).getEndGame() && aPersonnage.get(1).getEndGame() && aPersonnage.get(2).getEndGame() && aPersonnage.get(3).getEndGame()){
341            this.aGui.println("Félicitation, 4 personnes ont rejoins votre équipe, vous avez gagné !");
342            aPersonnage.get(0).setEndGame();
343        }
344    }//printLocationInfo()
345    
346    /**
347     * Méthode qui permet d'afficher les messages des personnages qui parle lorsqu'on arrive dans un lieu.
348     */
349    private void printFirstTalk()
350    {
351        if( !(this.aPlayer.getCurrentRoom().getPersonnageList().getFirstTalkPersonnageList()).isEmpty() ){
352            ArrayList<Personnage> vListPerso = this.aPlayer.getCurrentRoom().getPersonnageList().getFirstTalkPersonnageList();
353            for(Personnage vPerso : vListPerso)
354            {
355                this.aGui.println("\n" + vPerso.getFirstTalkString() );
356            }
357        }
358    }//printFirstTalk()

402    /**
403     * Méthode qui active le mode Timer.
404     */
405    public void timeAction()
406    {
407        this.aGui.println("Mode timer activé, vous avez droit à "+ this.aPlayer.getActionMax() +
408        " actions au maximum avant de perdre");
409        this.aPlayer.setAction(0);
410        this.aPlayer.setTimeMode(true);
411    }//timeAction()

413    /**
414     * Méthode qui affiche les informations sur le mode Time, ou met fin au jeu dû au manque de temp
415     */
416    public void timeEtat()
417    {
418        if(this.aPlayer.getTimeMode() ){
419            if( this.aPlayer.timeUp() ){
420                this.endTime();
421            }else{
422                this.aPlayer.plusUnAction();
423                this.printTimeInfo();
424            }
425        }
426    }//timeEtat()

428    /**
429     * Méthode qui exécute la/les classe de commande associé à la commande reçu en paramètre.
430     * @param pCommandLine, une commande en String
431     */
432    public void interpretCommand(final String pCommandLine)
433    {
434        this.aGui.println("> " + pCommandLine);
435        Command vCommande = this.aParser.getCommand(pCommandLine);
436        
437        if(vCommande == null){
438            this.aGui.println("Je ne comprend pas ce que je dois faire...");
439            return;
440        }
441
442        vCommande.execute(this.aPlayer, this.aGui, this);
443        
444    }//interpretCommand()
