266    /**
267     * Méthode qui permer d'ajouter un Item à l'inventaire du joueur.
268     * @param pNameItem, le nom de l'Item qu'on souhaite ajouter en String.
269     * @return retourne un String, qui décrit différent situation pour d'autre instruction.
270     */
271    public String ajouterInvItem(final String pNameItem)
272    {
273        if(this.aCurrentRoom.getItem( pNameItem ) == null){
274            return "inexistance";
275        }else{
276            int vPoid = this.aCurrentRoom.getItem( pNameItem ).getItemWeight();
277            if( this.aPoid + vPoid <= this.aPoidMax){
278                Item vObjet = this.aCurrentRoom.getItem( pNameItem );
279                this.aInvItems.ajouterItem( pNameItem, vObjet);
280                this.aPoid += vPoid;
281                this.aCurrentRoom.enleverItem( pNameItem );
282                return "ok";
283            }else{
284                return "lourd";
285            }
286        }
287    }//ajouterInvItem()
288    
289    /**
290     * Méthode permettant d'enlever un Item de l'inventaire et de le poser dans la Room actuel où se trouve
291     * le joueur
292     * @param pNameItem, le nom de l'Item qu'on souhaite enlever de l'inventaire
293     * @return retourne un String, qui décrit différent situation pour d'autre instruction.
294     */
295    public String enleverInvItem(final String pNameItem )
296    {
297        if( this.aInvItems.contientItem(pNameItem) ){
298            Item vObjet = this.aInvItems.getItem( pNameItem );
299            int vPoid = vObjet.getItemWeight();
300            this.aCurrentRoom.ajouterItem( pNameItem, vObjet);
301            this.aInvItems.enleverItem( pNameItem);
302            this.aPoid -= vPoid;
303            return "ok";
304        }else{
305            return "inexistance";
306        }
307    }//enleverInvItem()

331    /**
332     * Méthode qui permet de changer la room actuel du joueur et de le faire déplacer selon la direction donnée.
333     * @param pDirection, la direction que le joueur veut aller en String.
334     * @return true, si le joueur a pu se déplacer. False sinon
335     */
336    public boolean goRoom( final String pDirection )
337    {
338        Room vNextRoom = this.aCurrentRoom.getExit( pDirection );
339        if(vNextRoom == null){
340            return false;
341        }else{
342            this.aBackRoom.push(this.aCurrentRoom);
343            this.aCurrentRoom = vNextRoom;
344            return true;
345        }
346    }//goRoom()

383     */
384    public String tirerBeamer()
385    {
386        if( this.contientItem("beamer") ){
387            Beamer vBeamer = (Beamer)this.getInvItem("beamer");
388            if( !(vBeamer.getCharge() == null) ){
389                this.setCurrentRoom( vBeamer.tirer() );
390                return "ok";
391            }else{
392                return "unknown";
393            }
394        }else{
395            return "inexistance";
396        }
397    }//tirerBeamer()

428    public void DirectionDroite()
429    {
430        switch( this.aDirection ) {
431            case "N" :
432                this.aDirection = "E";
433                break;
434            case "E" :
435                this.aDirection = "S";
436                break;
437            case "S" :
438                this.aDirection = "W";
439                break;
440            case "W" :
441                this.aDirection = "N";
442                break;
443            default :
444                System.out.println("Alors là, il y a une erreur, j'arrive pas à changer la caméra Direction du joueur ?");
445                break;
446        }
447    }//DirectionDroite()
