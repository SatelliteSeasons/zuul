237    public void goRoom(final Command pCommande)
238    {
239        if(pCommande.hasSecondWord()==false){
240            this.aGui.println("Je dois me diriger où ?");
241            return;
242        }
243        String vDirection = pCommande.getSecondWord();
244        boolean vExistRoom = this.aPlayer.goRoom( vDirection );
245        
246        if(!vExistRoom){
247            this.aGui.println("Aucun passage vers ici...");
248            return;
249        }else{
250            if(this.aPlayer.getCurrentRoom().getImageName()!=null){
251                this.aGui.showImage(this.aPlayer.getCurrentRoom().getImageName());
252            }
253            printLocationInfo();
254            this.timeEtat();
255        }
256    }//goRoom()

314     * Méthode qui permet de manger des objets consommable.
315     * @param pCommande, une Command

318    {
319        if( !(pCommande.hasSecondWord()) ){
320            this.aGui.println("Je dois manger quoi ?");
321            return;
322        }
323        
324        String vObjet = pCommande.getSecondWord();
325        
326        if( vObjet.equals("cookie") ){
327            if( this.aPlayer.contientItem( vObjet ) ){
328                this.aPlayer.doublePoidMax();
329                this.aPlayer.enleverItemEtPoid( vObjet );
330                this.aGui.println("C'était pas si bon finalement, mais j'ai l'impression d'être devenu plus fort");
331                this.timeEtat();
332            }else{
333                this.aGui.println("Je n'ai pas cet objet en ma possession");
334            }
335        }else if( vObjet.equals("stoptime") ){
336            if( this.aPlayer.contientItem( vObjet ) ){
337                this.aPlayer.setTimeMode( false );
338                this.aPlayer.enleverItemEtPoid( vObjet );
339                this.aGui.println("Le bruit d'une aiguille s'est arrêté...");
340            }else{
341                this.aGui.println("Je n'ai pas cet objet en ma possession");
342            }
343        }else{
344            this.aGui.println("Je ne crois pas que ce truc se mange, et non je n'essayerai pas");
345        }
346    }//eat()

457    {
458        if( !(pCommande.hasSecondWord()) ){
459            this.aGui.println("Je dois utiliser quoi ?");
460            return;
461        }
462        String vNameItem = pCommande.getSecondWord();
463

467                if( vString.equals("inexistance") ){
468                    this.aGui.println("Je n'ai pas cet objet en ma possesion (tire)");
469                }else if( vString.equals("unknown") ){
470                    this.aGui.println("Arme non chargé");
471                }else{
472                    if(this.aPlayer.getCurrentRoom().getImageName()!=null){
473                        this.aGui.showImage(this.aPlayer.getCurrentRoom().getImageName());
474                    }
475                    printLocationInfo();
476                    this.timeEtat();

480                String vInstruction = this.aPlayer.chargerBeamer();
481                if( vInstruction.equals("ok") ){
482                    this.aGui.println("Je charge l'arme pour qu'il enregistre cet endroit");
483                }else{
484                    this.aGui.println("Je n'ai pas cet objet en ma possession");
485                }
486            }
487        }else{
488            this.aGui.println("Je n'ai aucune idée de quoi faire avec cet objet...");
489        }
490    }//utiliser()

492    /**
493     * fonction qui exécute la/les méthodes associé à la commande reçu en paramètre. Et vérifie si le
494     * joueurs souhaite quitter le jeu.
495     * @param pCommandLine, une commande en String
496     */
497    public void interpretCommand(final String pCommandLine)
498    {
499        this.aGui.println("> " + pCommandLine);
500        Command vCommande = this.aParser.getCommand(pCommandLine);
501        
502        
503        if(vCommande.isUnknown()){
504            this.aGui.println("I don't know what you mean...");
505            return;
506        }
507        
508        String vCommandeWord = vCommande.getCommandWord();
509        if(vCommandeWord.equals("quit")){
510            if(vCommande.hasSecondWord()){
511            this.aGui.println("Quit what ?");
512            }else{
513            this.endGame();
514            }
515        }else if(vCommandeWord.equals("go")){
516            this.goRoom(vCommande);
517        }else if(vCommandeWord.equals("help")){
518            this.printHelp();
519        }else if(vCommandeWord.equals("regarder")){   
520            this.look(vCommande);
521        }else if(vCommandeWord.equals("manger")){
522            this.eat(vCommande);
523        }else if(vCommandeWord.equals("back")){
524            this.back(vCommande);
525        }else if(vCommandeWord.equals("test")){
526            this.test(vCommande);
527        }else if(vCommandeWord.equals("prendre")){
528            this.prendre(vCommande);
529        }else if(vCommandeWord.equals("poser")){
530            this.poser(vCommande);
531        }else if(vCommandeWord.equals("inventaire")){
532            this.inventaire();
533        }else if(vCommandeWord.equals("testTime")){
534            this.timeAction();
535        }else if(vCommandeWord.equals("utiliser")){
536            this.utiliser(vCommande);
537        }else if(vCommandeWord.equals("alea")){
538            this.alea(vCommande);
539        }else{
540            this.aGui.println("Erreur du programmeur : commande non reconnue !");
541        }
542    }//interpretCommand()
