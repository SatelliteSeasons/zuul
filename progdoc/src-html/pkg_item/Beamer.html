001package pkg_item;
002
003import pkg_room.Room;
004
005/**
006 * Classe Beamer - Un Item spécial qui permet de mémoriser une Room et de le renvoyer.
007 *
008 * @author Richard Ho
009 * @version 03/04/2023
010 */
011public class Beamer extends Item
012{
013    private Room aCharge;
014    
015    /**
016     * Constructeur qui crée l'objet beamer avec une charge(aCharge) vide.
017     */
018    public Beamer()
019    {
020        super("beamer","Une machine capable de distordre le temps, permet lorsqu'il est chargé dans un lieu, d'y retourner depuis n'importe quelle espace, très puissant", 100);
021        this.aCharge = null;
022    }//Beamer()
023    
024    /**
025     * Méthode qui retourne la Room mémorisé de l'objet courant.
026     * @return Room
027     */
028    public Room getCharge()
029    {
030        return this.aCharge;
031    }//getCharge()
032    
033    /**
034     * Méthode boolean permettant de dire si l'objet courant a mémorisé ou non une Room.
035     * @return true, si une Room dans la mémoire. False sinon.
036     */
037    public boolean estCharger()
038    {
039        if( this.aCharge == null){
040            return false;
041        }else{
042            return true;
043        }
044    }//estCharger()
045    
046    /**
047     * Méthode permettant d'associer une Room dans la mémoire(attribut) de l'objet courant.
048     * @param pRoom, la Room qu'on souhaite mémoriser.
049     */
050    public void charger(final Room pRoom)
051    {
052        this.aCharge = pRoom;
053    }//charger()
054    
055    /**
056     * Méthode permettant de retourner la Room dans la mémoire et de le vider.
057     * @return Room
058     */
059    public Room tirer()
060    {
061        Room vRoom = this.aCharge;
062        this.aCharge = null;
063        return vRoom;
064    }//tirer()
065}//Beamer
