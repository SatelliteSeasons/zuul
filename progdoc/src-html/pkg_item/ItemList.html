001package pkg_item;
002
003
004import java.util.HashMap;
005import java.util.Set;
006import java.util.Collection;
007
008/**
009 * Classe ItemList - Permet de créer, de modifier et d'accéder à la liste d'Item.
010 *
011 * @author Richard Ho
012 * @version 03/04/2023
013 */
014public class ItemList
015{
016    private HashMap<String, Item> aItemList;
017    
018    /**
019     * Constructeur qui permet de déclarer une liste de valeur d'Item sous forme de HashMap avec leur clé en String.
020     */
021    public ItemList()
022    {
023        this.aItemList = new HashMap<String, Item>();
024    }//ItemList()
025    
026    /**
027     * Méthode pour ajouter un Item dans la liste de l'objet courant.
028     * @param pNameItem, est le nom/clé de l'Item.
029     * @param pItem, l'Item qu'on souhaite intégrer dans la liste.
030     */
031    public void ajouterItem( final String pNameItem , final Item pItem )
032    {
033        this.aItemList.put( pNameItem, pItem );
034    }//ajouterItem()
035    
036    /**
037     * Méthode pour enlever un Item dans la liste de l'objet courant.
038     * @param pNameItem, le nom/clé de l'Item.
039     */
040    public void enleverItem( final String pNameItem )
041    {
042        this.aItemList.remove( pNameItem );
043    }//enleverItem()
044    
045    /**
046     * Méthode qui retourne l'Item de la liste lorsqu'on donne le nom/clé de celui-ci.
047     * @param pNameItem, le nom/clé de l'Item.
048     * @return l'Item selon le nom/clé donné en paramètre.
049     */
050    public Item getItem( final String pNameItem )
051    {
052        return this.aItemList.get( pNameItem );
053    }//getItem()
054    
055    /**
056     * Méthode qui retourne un booléen selon si la liste de l'objet courant est vide ou non.
057     * @return true, si l'objet courant, possède une liste vide.
058     */
059    public boolean empty()
060    {
061        return this.aItemList.isEmpty();
062    }//empty()
063    
064    /**
065     * Méthode qui retourne un booléen selon si la liste contient ou non le nom de l'Item passé en paramètre.
066     * @param pNameItem, le nom de l'Item en String qu'on souhaite savoir si il se trouve dans la liste.
067     * @return true, si l'Item est bien présent.
068     */
069    public boolean contientItem( final String pNameItem )
070    {
071        return this.aItemList.containsKey( pNameItem );
072    }//contientItem()
073    
074    /**
075     * Méthode qui retourne un ensemble de String, qui représente tout les noms des Items de la liste de l'objet courant.
076     * @return Un ensemble de String (Set de type String).
077     */
078    public Set<String> keySetItem()
079    {
080        return this.aItemList.keySet();
081    }//keySetItem()
082    
083    /**
084     * Méthode qui retourne la liste des Items présent dans la liste de l'objet courant sous forme de String.
085     * @return String
086     */
087    public String itemsString()
088    {
089        String vInventaire = "";
090        for( Item vObjet : this.aItemList.values() )
091        {
092            vInventaire += " " + vObjet.getItemName() + ",";
093        }
094        return vInventaire;
095    }//itemsString()
096}//ItemList
