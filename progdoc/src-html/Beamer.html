002/**
003 * Classe Beamer - Un Item spécial qui permet de mémoriser une Room et de le renvoyer.
004 *
005 * @author Richard Ho
006 * @version 03/04/2023
007 */
008public class Beamer extends Item
009{
010    private Room aCharge;
011    
012    /**
013     * Constructeur qui crée l'objet beamer avec une charge(aCharge) vide.
014     */
015    public Beamer()
016    {
017        super("beamer","Une machine capable de distordre le temps, permet lorsqu'il est chargé dans un lieu, d'y retourner depuis n'importe quelle espace, très puissant", 100);
018        this.aCharge = null;
019    }//Beamer()
020    
021    /**
022     * Méthode qui retourne la Room mémorisé de l'objet courant.
023     * @return Room
024     */
025    public Room getCharge()
026    {
027        return this.aCharge;
028    }//getCharge()
029    
030    /**
031     * Méthode boolean permettant de dire si l'objet courant a mémorisé ou non une Room.
032     * @return true, si une Room dans la mémoire. False sinon.
033     */
034    public boolean estCharger()
035    {
036        if( this.aCharge == null){
037            return false;
038        }else{
039            return true;
040        }
041    }//estCharger()
042    
043    /**
044     * Méthode permettant d'associer une Room dans la mémoire(attribut) de l'objet courant.
045     * @param pRoom, la Room qu'on souhaite mémoriser.
046     */
047    public void charger(final Room pRoom)
048    {
049        this.aCharge = pRoom;
050    }//charger()
051    
052    /**
053     * Méthode permettant de retourner la Room dans la mémoire et de le vider.
054     * @return Room
055     */
056    public Room tirer()
057    {
058        Room vRoom = this.aCharge;
059        this.aCharge = null;
060        return vRoom;
061    }//tirer()
062}//Beamer
