001import java.util.Random;
002import java.util.ArrayList;
003
004/**
005 * Classe RoomRandomizer - Une classe qui a pour but de générer une room de manière (pseudo) aléatoire
006 * et de le renvoyer.
007 *
008 * @author Richard Ho
009 * @version 03/04/2023
010 */
011public class RoomRandomizer
012{
013    private Random aRandom;
014    private ArrayList<Room> aRoomsList;
015    
016    /**
017     * Constructeur naturel qui récupère une liste de Room qu'on va exploiter.
018     * (Seed pour débug = 10)
019     * @param pRoomRandomList, une liste de Room qu'on souhaite piocher au hasard.
020     */
021    public RoomRandomizer(final ArrayList<Room> pRoomRandomList)
022    {
023        this.aRandom = new Random();
024        this.aRoomsList = pRoomRandomList;
025    }//RoomRandomizer()
026    
027    /**
028     * Méthode qui retourne de manière aléatoire une Room depuis une liste de Room.
029     * @return Room
030     */
031    public Room getRoom()
032    {
033        return this.aRoomsList.get( this.aRandom.nextInt( this.aRoomsList.size() ) );
034    }//getRoom()
035    
036    /**
037     * Méthode qui retourne différent Room selon leur nom.
038     * @param pSalle, le nom de la Room en String qu'on souhaite retourner.
039     * @return la Room.
040     */
041    public Room getRoomAlea(final String pSalle)
042    {
043        String vName = pSalle;
044        for (Room vRoom : this.aRoomsList)
045        {
046            if(vName.equals( vRoom.getName() ) ){
047                return vRoom;
048            }
049        }
050        return null;
051    }//getRoomAlea()
052    
053}//RoomRandomizer
