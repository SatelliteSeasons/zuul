025import  java.util.HashMap; 
026import  java.util.Iterator;

028/**
029 * This class is the main class of the "World of Zuul" application. 
030 * "World of Zuul" is a very simple, text based adventure game.
031 * 
032 * This class holds a collection of all command words known to the game.
033 * It is used to recognise commands as they are typed in.
034 *
035 * @author Michael Kolling and David J. Barnes + Edited by Richard Ho
036 * @version 2011.07.31 + 2023.04.09
037 */
038public class CommandWords
039{
040    private final HashMap<String, Command> aValidCommands;
041
042    /**
043     * Constructor - initialise the command words.
044     */
045    public CommandWords()
046    {
047        this.aValidCommands = new HashMap<String, Command>();
048        this.aValidCommands.put("go", new GoCommand() );
049        this.aValidCommands.put("aide", new HelpCommand() );
050        this.aValidCommands.put("quitter", new QuitCommand() );
051        this.aValidCommands.put("regarder", new LookCommand() );
052        this.aValidCommands.put("manger", new EatCommand() );
053        this.aValidCommands.put("back", new BackCommand() );
054        this.aValidCommands.put("prendre", new TakeCommand() );
055        this.aValidCommands.put("poser", new PutCommand() );
056        this.aValidCommands.put("inventaire", new InventoryCommand() );
057        this.aValidCommands.put("utiliser", new UseCommand() );
058        this.aValidCommands.put("alea", new AleaCommand() );
059        this.aValidCommands.put("testTime", new TestTimeCommand() );
060        this.aValidCommands.put("test", new TestCommand() );
061    
062        this.aValidCommands.put("parler", new TalkCommand() );
063        this.aValidCommands.put("donner", new GiveCommand() );
064        
065        this.aValidCommands.put("droite", new DroiteCommand() );
066        this.aValidCommands.put("gauche", new GaucheCommand() );
067        
068        this.aValidCommands.put("avancer", new AvancerCommand() );
069        this.aValidCommands.put("derriere", new DerriereCommand() );
070    } // CommandWords()
071    
072    /**
073     * Méthode qui récupère en paramètre la commande en String, et qui renvoie la classe de la commande correspondant.
074     * @param pWord La commande en premier mot.
075     * @return Command, la classe de commande correspondant.
076     */
077    public Command get( String pWord )
078    {
079        return (Command)this.aValidCommands.get( pWord );
080    }//get()
081    
082    /**
083     * Check whether a given String is a valid command word. 
084     * @param pString, the word we want to verify.
085     * @return true if a given string is a valid command,
086     * false if it isn't.
087     */
088    public boolean isCommand( final String pString )
089    {
090        for(Iterator i = this.aValidCommands.keySet().iterator(); i.hasNext(); ) {
091            if( pString.equals( i.next() ) ){
092                return true;
093            }
094        }
095        return false;
096    } // isCommand()
097    
098    /**
099     * Méthode qui retourne la liste de commande possible sous forme de String.
100     * @return String
101     */
102    public String getCommandList()
103    {
104        String vLocal="";
105        for(Iterator i = this.aValidCommands.keySet().iterator(); i.hasNext(); ) {
106            vLocal += i.next() + " ";
107        }
108        return vLocal;
109    }// getCommandList()
110    
111}// CommandWords
