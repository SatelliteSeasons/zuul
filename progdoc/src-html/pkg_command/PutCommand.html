001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007/**
008 * Classe PutCommand - Représentation des instructions de la commande "poser"
009 *
010 * @author Richard Ho
011 * @version 09/04/2023
012 */
013public class PutCommand extends Command
014{
015    
016
017    /**
018     * Constructeur d'objets de classe PutCommand.
019     */
020    public PutCommand()
021    {
022        
023    }//PutCommand()
024
025    /**
026     * Méthode qui exécute une instruction correspondant à la commande "poser".
027     * @param pPlayer Player, le joueur.
028     * @param pGui UserInterface, l'interface.
029     * @param pGameEngine GameEngine, le moteur du jeu.
030     */
031    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine )
032    {
033        if( !(this.hasSecondWord()) ){
034            pGui.println("Je dois poser quoi ?");
035            return false; //optio
036        }
037        String vNameItem = this.getSecondWord();
038        String vExistItem = pPlayer.enleverInvItem( vNameItem );
039        
040        if( vExistItem.equals("ok") ){
041            pGui.println("Je pose l'objet suivant : " + vNameItem);
042            pGameEngine.timeEtat();
043        }else{
044            pGui.println("J'ai aucun objet du nom de " + vNameItem + " sur moi");
045        }
046        return false; //optio
047    }//execute()
048}//PutCommand
