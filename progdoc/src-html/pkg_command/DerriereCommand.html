001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007/**
008 * Classe DerriereCommand - Représentation des instructions de la commande "derriere"
009 *
010 * @author Richard Ho
011 * @version 22/05/2023
012 */
013public class DerriereCommand extends Command
014{
015
016    /**
017     * Constructeur d'objets de classe DerriereCommand
018     */
019    public DerriereCommand()
020    {
021    
022    }//DerriereCommand()
023
024    /**
025     * Méthode qui exécute une instruction correspondant à la commande "derriere".
026     * @param pPlayer Player, le joueur.
027     * @param pGui UserInterface, l'interface.
028     * @param pGameEngine GameEngine, le moteur du jeu.
029     */
030    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine )
031    {
032        if(this.hasSecondWord()){
033            pGui.println("Pas plus de mot");
034            return false; //optio
035        }
036        pPlayer.DirectionDerriere();
037        pGui.showImage( pPlayer.getCurrentRoom().getImageHashMap().getImage( pPlayer.getDirection() ) );
038        return false; //optio
039        
040    }//execute()
041}//DerriereCommand
