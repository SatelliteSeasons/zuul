001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007/**
008 * Classe TakeCommand - Représentation des instructions de la commande "prendre"
009 *
010 * @author Richard Ho
011 * @version 09/04/2023
012 */
013public class TakeCommand extends Command
014{
015
016    /**
017     * Constructeur d'objets de classe TakeCommand
018     */
019    public TakeCommand()
020    {
021    }//TakeCommand()
022
023    /**
024     * Méthode qui exécute une instruction correspondant à la commande "prendre".
025     * @param pPlayer Player, le joueur.
026     * @param pGui UserInterface, l'interface.
027     * @param pGameEngine GameEngine, le moteur du jeu.
028     */
029    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine)
030    {
031        if( !(this.hasSecondWord()) ){
032            pGui.println("Je dois prendre quoi ?");
033            return false; //optio
034        }
035        String vNameItem = this.getSecondWord();
036        String vExistItem = pPlayer.ajouterInvItem( vNameItem );
037        if( vExistItem.equals( "ok" ) ){
038            pGui.println("Je prend l'objet suivant : " + vNameItem);
039            pGameEngine.timeEtat();
040        }else if( vExistItem.equals("lourd") ){
041            pGui.println("L'objet est trop lourd pour moi");
042        }else{
043            pGui.println("Il n'y a aucun objet du nom de " + vNameItem + " ici");
044        }
045        return false; //optio
046    }//execute()
047}//TakeCommand
