001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007/**
008 * Classe EatCommand - Représentation des instructions de la commande "eat"
009 *
010 * @author Richard Ho
011 * @version 09/04/2023
012 */
013public class EatCommand extends Command
014{
015
016    /**
017     * Constructeur d'objets de classe EatCommand
018     */
019    public EatCommand()
020    {
021    }//EatCommand()
022
023    /**
024     * Méthode qui exécute une instruction correspondant à la commande "manger".
025     * @param pPlayer Player, le joueur.
026     * @param pGui UserInterface, l'interface.
027     * @param pGameEngine GameEngine, le moteur du jeu.
028     */
029    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine)
030    {
031        if( !(this.hasSecondWord()) ){
032            pGui.println("Je dois manger quoi ?");
033            return false; //optio
034        }
035        
036        String vObjet = this.getSecondWord();
037        
038        if( vObjet.equals("cookie") ){
039            if( pPlayer.contientItem( vObjet ) ){
040                pPlayer.doublePoidMax();
041                pPlayer.enleverItemEtPoid( vObjet );
042                pGui.println("C'était pas si bon finalement, mais j'ai l'impression d'être devenu plus fort");
043                pGameEngine.timeEtat();
044            }else{
045                pGui.println("Je n'ai pas cet objet en ma possession");
046            }
047        }else if( vObjet.equals("stoptime") ){
048            if( pPlayer.contientItem( vObjet ) ){
049                pPlayer.setTimeMode( false );
050                pPlayer.enleverItemEtPoid( vObjet );
051                pGui.println("Le bruit d'une aiguille s'est arrêté...");
052            }else{
053                pGui.println("Je n'ai pas cet objet en ma possession");
054            }
055        }else{
056            pGui.println("Je ne crois pas que ce truc se mange, et non je n'essayerai pas");
057        }
058        return false; //optio
059    }//execute()
060}//EatCommand
