001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007import pkg_character.Personnage;
008
009
010/**
011 * Classe TalkCommand - Représentation des instructions de la commande "parler"
012 *
013 * @author Richard
014 * @version 22/05/2023
015 */
016public class TalkCommand extends Command
017{
018
019    /**
020     * Constructeur d'objets de classe TalkCommand
021     */
022    public TalkCommand()
023    {
024    }//TalkCommand()
025
026    /**
027     * Exécute la commande. Méthode qui retourne un indicateur de si le jeu doit se terminer ou non
028     * @param pPlayer Un joueur en Player.
029     * @param pGui Un interface UserInterface.
030     * @param pGameEngine Un moteur de jeu GameEngine.
031     * @return true, si le jeu doit se terminer. False sinon.
032     */
033    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine)
034    {
035        if( !(this.hasSecondWord()) ){
036            pGui.println( "Je dois parler à qui ?" );
037            return false; //optio
038        }
039        
040        String vPerso = this.getSecondWord();
041        
042        if( pPlayer.getCurrentRoom().contientPersonnage(vPerso) )
043        {
044            Personnage vPersonnage = pPlayer.getCurrentRoom().getPersonnage(vPerso);
045            pGui.println( vPersonnage.talkString() );
046        }else{
047            pGui.println("Il n'y a personne de ce nom ici.");
048        }
049        return false; //optio
050    }//execute()
051    
052}//TalkCommand
