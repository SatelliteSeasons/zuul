001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007/**
008 * Classe Command - Permet de définir et diviser un texte en deux mot. Et représente
009 * une commande du jeu.
010 *
011 * @author Richard Ho
012 * 
013 * @version 22/05/2023
014 */
015public abstract class Command
016{
017    private String aSecondWord;
018    private String aThirdWord;
019    
020    /**
021     * Constructeur par défaut. (A compléter) 
022     */
023    public Command()
024    {
025        this.aSecondWord = null;
026        this.aThirdWord = null;
027    }//Command()
028    
029    /**
030     * Méthode qui retourne le deuxième mot de la commande en String.
031     * @return String
032     */
033    public String getSecondWord()
034    {
035        return this.aSecondWord;
036    }//getSecondWord()
037    
038    /**
039     * Méthode boolean qui retourne si il y a existence d'une second mot de l'objet courant.
040     * @return true, si la commande possède un deuxième mot. False sinon
041     */
042    public boolean hasSecondWord()
043    {
044        return this.aSecondWord!=null;
045    }//hasSecondWord()
046    
047    /**
048     * Méthode qui modifie le second mot de la commande.
049     * @param pSecondWord, le second mot en String
050     */
051    public void setSecondWord( final String pSecondWord )
052    {
053        this.aSecondWord = pSecondWord;
054    }//setSecondWord()
055    
056    //Troisième mot :
057    /**
058     * Méthode qui retourne le troisième mot de la commande en String.
059     * @return String
060     */
061    public String getThirdWord()
062    {
063        return this.aThirdWord;
064    }//getThirdWord()
065    
066    /**
067     * Méthode boolean qui retourne si il y a existence d'un troisème mot de l'objet courant.
068     * @return true, si la commande possède un troisième mot. False sinon
069     */
070    public boolean hasThirdWord()
071    {
072        return this.aThirdWord != null;
073    }//hasThirdWord()
074    
075    /**
076     * Méthode qui modifie le troisième mot de la commande.
077     * @param pThirdWord, le troisième mot en String
078     */
079    public void setThirdWord( final String pThirdWord )
080    {
081        this.aThirdWord = pThirdWord;
082    }//setThirdWord()
083    
084    //Abstract method :
085    
086    /**
087     * Exécute la commande. Méthode qui retourne un indicateur de si le jeu doit se terminer ou non
088     * @param pPlayer Un joueur en Player.
089     * @param pGui Un interface UserInterface.
090     * @param pGameEngine Un moteur de jeu GameEngine.
091     * @return true, si le jeu doit se terminer. False sinon.
092     */
093    public abstract boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine );
094    //execute()
095} // Command
