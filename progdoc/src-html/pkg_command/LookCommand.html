001package pkg_command;
002
003import pkg_system.Player;
004import pkg_system.UserInterface;
005import pkg_system.GameEngine;
006
007import pkg_item.Item;
008
009/**
010 * Classe LookCommand - Représentation des instructions de la commande "regarder"
011 *
012 * @author Richard Ho
013 * @version 09/04/2023
014 */
015public class LookCommand extends Command
016{
017    /**
018     * Constructeur d'objets de classe LookCommand
019     */
020    public LookCommand()
021    {
022    }//LookCommand()
023
024    /**
025     * Méthode qui exécute une instruction correspondant à la commande "regarder".
026     * @param pPlayer Player, le joueur.
027     * @param pGui UserInterface, l'interface.
028     * @param pGameEngine GameEngine, le moteur du jeu.
029     */
030    @Override public boolean execute( final Player pPlayer, final UserInterface pGui, final GameEngine pGameEngine )
031    {
032        if(this.hasSecondWord()){
033            String vSecondMot = this.getSecondWord();
034            Item vObjetActuel = ( pPlayer.getCurrentRoom().getItem( vSecondMot ) );
035            if(vObjetActuel != null && ( vObjetActuel.getItemName().equals( vSecondMot ) ) )
036            {      
037                pGui.println(vObjetActuel.getLongItemDescription());
038            }else{
039                pGui.println("''" + this.getSecondWord() + "''" +"..." + " Ce que je cherche n'est probablement pas ici");
040            }
041        }else{
042            pGui.println( pPlayer.getCurrentRoom().getLongDescription() );
043        }
044        return false; //optio
045    }//execute()
046}//LookCommand
